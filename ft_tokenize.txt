# Pseudo Code for tokenization :

function lexer(input: string) -> array of tokens
    tokens = empty array of tokens
    current_token = empty string
    inside_quotes = false
    escaped = false

    for each character in input
        if character is escape character '\' and not escaped
            escaped = true
            append character to current_token
        else if character is double quote '"' and not escaped
            inside_quotes = not inside_quotes
            append character to current_token
        else if character is space and not inside_quotes and not escaped
            if current_token is not empty
                add current_token to tokens
                current_token = empty string
        else if character is special character and not inside_quotes and not escaped
            if current_token is not empty
                add current_token to tokens
                current_token = empty string
            add character as token to tokens
        else
            append character to current_token
            escaped = false

    if current_token is not empty
        add current_token to tokens

    return tokens
